class CaRhythmScoringEngine:
    """
    CaRhythm Assessment Scoring Engine
    ===================================
    Scores user responses across 3 modules:
      1. RIASEC (Likert + Forced Choice + Ranking)
      2. Big Five Personality
      3. Behavioral Traits

    Input: User responses in dict format {item_id: response_value}
    Output: Structured "Rhythm Profile" with strength labels, behavioral flags, and Ikigai zones.

    Designed for web/mobile apps. Zero dependencies.
    """

    def __init__(self):
        # Load scoring thresholds from JSON spec
        self.strength_thresholds = {
            "RIASEC": {"low": 6, "medium": 10, "high": 13, "very_high": 15},
            "BigFive": {"low": 10, "medium": 16, "high": 21, "very_high": 25},
            "Behavioral": {"low": 6, "medium": 10, "high": 13, "very_high": 15}
        }
        
        # Initialize domains/traits
        self.holland_domains = ["R", "I", "A", "S", "E", "C"]
        self.bigfive_traits = ["O", "C", "E", "A", "N"]  # N = Emotional Stability
        self.behavioral_traits = [
            "motivation_type", "grit_persistence", "self_efficacy", 
            "resilience", "learning_orientation", "empathy", "task_start_tempo"
        ]

    def score_assessment(self, responses: dict) -> dict:
        """
        Main entry point. Scores full assessment and returns Rhythm Profile.
        
        Args:
            responses: Dict of {item_id: response_value}
                       e.g., {"R1": 5, "FC_RI_1": "A", "RANK1": ["S", "A", "E", "I", "R", "C"]}
        
        Returns:
            Dict with keys: riasec_scores, bigfive_scores, behavioral_scores, 
                           ikigai_zones, behavioral_flags, rhythm_profile
        """
        # Initialize scores
        riasec_scores = {domain: 0 for domain in self.holland_domains}
        bigfive_scores = {trait: 0 for trait in self.bigfive_traits}
        behavioral_scores = {trait: 0 for trait in self.behavioral_traits}
        
        # Initialize behavioral flags
        behavioral_flags = {
            "procrastination_risk": 0,
            "perfectionism_risk": 0,
            "low_grit_risk": 0,
            "poor_regulation_risk": 0,
            "impulse_control_issue": 0,
            "growth_mindset": 0
        }

        # Score each module
        self._score_riasec(responses, riasec_scores, behavioral_flags)
        self._score_bigfive(responses, bigfive_scores, behavioral_flags)
        self._score_behavioral(responses, behavioral_scores, behavioral_flags)

        # Convert to strength labels
        riasec_strengths = self._convert_to_strength_labels(riasec_scores, "RIASEC")
        bigfive_strengths = self._convert_to_strength_labels(bigfive_scores, "BigFive")
        behavioral_strengths = self._convert_to_strength_labels(behavioral_scores, "Behavioral")

        # Generate Ikigai zones
        ikigai_zones = self._generate_ikigai_zones(riasec_strengths, bigfive_strengths, behavioral_strengths, responses)

        # Generate final rhythm profile
        rhythm_profile = self._generate_rhythm_profile(
            riasec_strengths, bigfive_strengths, behavioral_strengths, ikigai_zones, behavioral_flags
        )

        return {
            "riasec_scores": riasec_scores,
            "bigfive_scores": bigfive_scores,
            "behavioral_scores": behavioral_scores,
            "riasec_strengths": riasec_strengths,
            "bigfive_strengths": bigfive_strengths,
            "behavioral_strengths": behavioral_strengths,
            "ikigai_zones": ikigai_zones,
            "behavioral_flags": behavioral_flags,
            "rhythm_profile": rhythm_profile,
            "meta": {
                "version": "1.1",
                "scoring_engine": "CaRhythm v1.1"
            }
        }

    def _score_riasec(self, responses: dict, scores: dict, flags: dict):
        """Score RIASEC module (Likert + Forced Choice + Ranking)."""
        
        # LIKERT ITEMS (R1-R3, I1-I3, A1-A3, S1-S3, E1-E3, C1-C3)
        likert_items = [f"{domain}{i}" for domain in self.holland_domains for i in range(1,4)]
        for item_id in likert_items:
            if item_id in responses:
                try:
                    score_val = int(responses[item_id])
                    # Extract domain from item_id (first letter)
                    domain = item_id[0]
                    if domain in scores:
                        scores[domain] += score_val
                except (ValueError, TypeError):
                    continue  # Skip invalid responses

        # FORCED CHOICE ITEMS
        forced_choice_map = {
            "FC_RI_1": {"A": "R", "B": "I"},
            "FC_IA_1": {"A": "I", "B": "A"},
            "FC_AS_1": {"A": "A", "B": "S"},
            "FC_SE_1": {"A": "S", "B": "E"},
            "FC_EC_1": {"A": "E", "B": "C"},
            "FC_CR_1": {"A": "C", "B": "R"}
        }
        
        for item_id, choice_map in forced_choice_map.items():
            if item_id in responses:
                choice = responses[item_id]
                if choice in choice_map:
                    domain = choice_map[choice]
                    scores[domain] += 3  # Fixed 3 points per forced choice

        # RANKING SETS
        ranking_sets = ["RANK1", "RANK2", "RANK3"]
        position_points = {1: 6, 2: 5, 3: 4, 4: 3, 5: 2, 6: 1}
        
        for rank_set in ranking_sets:
            if rank_set in responses:
                ranked_list = responses[rank_set]  # List of domains in order [1st, 2nd, ..., 6th]
                for position, domain in enumerate(ranked_list, 1):
                    points = position_points.get(position, 0)
                    if domain in scores:
                        scores[domain] += points

    def _score_bigfive(self, responses: dict, scores: dict, flags: dict):
        """Score Big Five module."""
        
        # Map item IDs to traits
        bigfive_item_map = {
            "O1": "O", "O2": "O", "O3": "O", "O4": "O", "O5": "O",
            "C1": "C", "C2": "C", "C3": "C", "C4": "C", "C5": "C",
            "E1": "E", "E2": "E", "E3": "E", "E4": "E", "E5": "E",
            "A1": "A", "A2": "A", "A3": "A", "A4": "A", "A5": "A",
            "N1": "N", "N2": "N", "N3": "N", "N4": "N", "N5": "N"
        }
        
        for item_id, trait in bigfive_item_map.items():
            if item_id in responses:
                try:
                    score_val = int(responses[item_id])
                    # Handle reverse-scored items (N1-N5)
                    if item_id in ["N1", "N2", "N3", "N4", "N5"]:
                        score_val = 6 - score_val  # Flip 1‚Üî5, 2‚Üî4, 3‚Üî3
                    scores[trait] += score_val
                except (ValueError, TypeError):
                    continue

    def _score_behavioral(self, responses: dict, scores: dict, flags: dict):
        """Score Behavioral Traits module."""
        
        # Map item IDs to traits
        behavioral_item_map = {
            "M1": "motivation_type", "M2": "motivation_type", "M3": "motivation_type",
            "G1": "grit_persistence", "G2": "grit_persistence", "G3": "grit_persistence",
            "SE1": "self_efficacy", "SE2": "self_efficacy", "SE3": "self_efficacy",

            "EM1": "empathy", "EM2": "empathy", "EM3": "empathy",
            "T1": "task_start_tempo", "T2": "task_start_tempo", "T3": "task_start_tempo"
        }
        
        for item_id, trait in behavioral_item_map.items():
            if item_id in responses:
                try:
                    score_val = int(responses[item_id])
                    # Handle reverse-scored items
                    if item_id in ["M3", "G3", "SE3", "R3", "L3", "T1", "T3"]:
                        score_val = 6 - score_val
                    scores[trait] += score_val
                    
                    # Set behavioral flags based on specific items
                    if item_id == "T1" and score_val <= 2:  # High procrastination risk
                        flags["procrastination_risk"] += 1
                    if item_id == "L3" and score_val <= 2:  # High perfectionism risk
                        flags["perfectionism_risk"] += 1
                    if item_id == "G3" and score_val <= 2:  # Low grit risk
                        flags["low_grit_risk"] += 1
                    if item_id == "R3" and score_val <= 2:  # Poor regulation risk
                        flags["poor_regulation_risk"] += 1
                    if item_id == "L1" and score_val >= 4:  # Growth mindset
                        flags["growth_mindset"] += 1
                        
                except (ValueError, TypeError):
                    continue

    def _convert_to_strength_labels(self, scores: dict, category: str) -> dict:
        """Convert raw scores to strength labels based on thresholds."""
        thresholds = self.strength_thresholds[category]
        strengths = {}
        
        for key, score in scores.items():
            if score <= thresholds["low"]:
                strengths[key] = "Low"
            elif score <= thresholds["medium"]:
                strengths[key] = "Medium"
            elif score <= thresholds["high"]:
                strengths[key] = "High"
            else:
                strengths[key] = "Very High"
                
        return strengths
